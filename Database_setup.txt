CREATE DATABASE LovelyPawsShelter;

USE LovelyPawsShelter;

CREATE TABLE admin (
  username VARCHAR(20) NOT NULL PRIMARY KEY COLLATE utf8_general_ci,
  password VARCHAR(20) NOT NULL COLLATE utf8_general_ci
);

CREATE TABLE adoption (
  email VARCHAR(30) NOT NULL COLLATE utf8_general_ci,
  petID int(10) NOT NULL,
  fname VARCHAR(20) NOT NULL COLLATE utf8_general_ci,
  lname VARCHAR(20) NOT NULL COLLATE utf8_general_ci,
  phoneNum VARCHAR(10) NOT NULL,
  statement VARCHAR(2000) NOT NULL COLLATE utf8_general_ci,
  PRIMARY KEY (email, petID)
);

CREATE TABLE animal (
  petID int(10) NOT NULL PRIMARY KEY,
  name VARCHAR(10) NOT NULL COLLATE utf8_general_ci,
  breed VARCHAR(20) NOT NULL COLLATE utf8_general_ci,
  age int(2) NOT NULL,
  gender VARCHAR(6) NOT NULL COLLATE utf8_general_ci,
  description VARCHAR(500) NOT NULL COLLATE utf8_general_ci
);

CREATE TABLE appointment (
  email VARCHAR(30) NOT NULL COLLATE utf8_general_ci,
  fname VARCHAR(20) NOT NULL COLLATE utf8_general_ci,
  lname VARCHAR(20) NOT NULL COLLATE utf8_general_ci,
  date date NOT NULL,
  time VARCHAR(2) NOT NULL COLLATE utf8_general_ci,
  PRIMARY KEY(email, date, time)
);
